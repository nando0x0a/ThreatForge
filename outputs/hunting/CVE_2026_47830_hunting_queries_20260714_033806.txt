### Query 3 — File System Write Artefacts in Sensitive Directories
*Target: Webshell drops, staged payloads, or persistence artefacts written post-exploitation.*

event_simpleName=NewExecutableWritten
| search TargetDirectoryName IN (
    "*\\inetpub\\wwwroot\\*",
    "*\\Temp\\*",
    "*\\AppData\\Roaming\\*",
    "*/var/www/*",
    "*/tmp/*",
    "*/dev/shm/*"
  )
| eval WriteTime=strftime(_time, "%Y-%m-%d %H:%M:%S")
| stats
    count AS WriteCount,
    values(TargetFileName) AS FilesWritten,
    values(TargetDirectoryName) AS Directories,
    values(SHA256HashData) AS Hashes,
    values(UserName) AS UserAccounts,
    min(WriteTime) AS FirstSeen,
    max(WriteTime) AS LastSeen
  by ComputerName, ParentBaseFileName
| where WriteCount >= 1
| sort -WriteCount
