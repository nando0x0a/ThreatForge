| search
    (
        FileName IN (
            "whoami.exe", "net.exe", "net1.exe", "ipconfig.exe",
            "nltest.exe", "wmic.exe", "powershell.exe", "cmd.exe",
            "certutil.exe", "bitsadmin.exe", "mshta.exe", "rundll32.exe",
            "regsvr32.exe", "wscript.exe", "cscript.exe"
        )
    )
    AND ParentBaseFileName NOT IN (
        "explorer.exe", "services.exe", "svchost.exe",
        "userinit.exe", "sihost.exe"
    )
| eval SuspicionFlag=case(
    FileName IN ("certutil.exe","bitsadmin.exe","mshta.exe"), "HIGH",
    FileName IN ("powershell.exe","rundll32.exe","regsvr32.exe"), "MEDIUM",
    true(), "LOW"
  )
| stats
    count            AS ExecCount,
    values(FileName) AS ChildProcesses,
    values(CommandLine) AS CommandLines,
    values(SuspicionFlag) AS SuspicionLevels,
    dc(ComputerName) AS HostCount
    by ParentBaseFileName, UserName, ComputerName
| where ExecCount > 2 OR SuspicionLevels="HIGH"
| sort -ExecCount
